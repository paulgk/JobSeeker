<!-- verifySession return shape — from src/lib/dal.ts lines 21-25 -->
```typescript
export const verifySession = cache(async () => {
  const session = await auth.api.getSession({ headers: await headers() })
  if (!session) redirect('/sign-in')
  return { isAuth: true as const, userId: session.user.id, user: session.user }
  // userId is string (better-auth text PK)
})
```

<!-- Current auth guard pattern in all three routes (does NOT capture userId) -->
```typescript
try {
  await verifySession()
} catch {
  return new Response(null, { status: 401 })
}
```

<!-- Target pattern (captures userId for use downstream) -->
```typescript
let userId: string
try {
  const session = await verifySession()
  userId = session.userId
} catch {
  return new Response(null, { status: 401 })
}
```

<!-- Why this pattern and not destructuring at the top: verifySession() calls redirect() which throws
     a Next.js redirect exception when the session is missing. The catch block converts that to a 401.
     If verifySession() is called outside the try block, the redirect exception bubbles uncaught → 500. -->
